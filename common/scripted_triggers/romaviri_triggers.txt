ep3_should_use_byzantine_graphics_trigger = {
    culture = { has_building_gfx = byzantine_building_gfx }
}

is_romaviri_emperor_trigger = {
    has_title = title:h_romaviri_empire
    has_title = title:h_eastern_romaviri_empire
}

is_romaviri_emperor_primary_title_trigger = {
    OR = {
        has_title = title:h_romaviri_empire
        has_title = title:h_eastern_romaviri_empire
    }
}

valid_for_romaviri_emperor_content_trigger = {
    has_ep3_dlc_trigger = yes
    culture = {
        has_cultural_pillar = heritage_elf
    }
    OR = {
        is_romaviri_emperor_primary_title_trigger = yes
        AND = {
            culture = {
                OR = {
                    this = culture:culture_elf
                    this = culture:elf_culture_serelion
                    this = culture:elf_culture_gwynthorn
                    this = culture:elf_culture_daelurin
                    this = culture:elf_culture_thundarael
                    this = culture:elf_culture_valerith
                    this = culture:elf_culture_lormelis
                    any_parent_culture_or_above = { this = culture:culture_elf }
                    any_parent_culture_or_above = { this = culture:elf_culture_serelion }
                    any_parent_culture_or_above = { this = culture:elf_culture_gwynthorn }
                    any_parent_culture_or_above = { this = culture:elf_culture_daelurin }
                    any_parent_culture_or_above = { this = culture:elf_culture_thundarael }
                    any_parent_culture_or_above = { this = culture:elf_culture_valerith }
                    any_parent_culture_or_above = { this = culture:elf_culture_lormelis }
                }
            }
            highest_held_title_tier >= tier_empire
        }
    }
}

has_any_charioteer_trait = {
    OR = {
        has_trait = charioteer_blue
        has_trait = charioteer_green
        has_trait = charioteer_white
        has_trait = charioteer_red
    }
}

available_charioteers_spots_trigger = {
    scope:activity = {
        any_guest_subset = {
            name = charioteers
            has_trait = charioteer_$TEAM$
            count < scope:activity.var:num_$TEAM$s
        }
    }
}
