            root.top_liege ?= { has_character_flag = $FLAG$ }

                has_variable = years_as_varangian
                has_character_flag = $FLAG$
                is_landed = no

            root.top_liege ?= { has_character_flag = $FLAG$ }
            AND = {
                has_variable = years_as_highguardian
                has_character_flag = $FLAG$
                is_landed = no
            }
        }
    }
}
