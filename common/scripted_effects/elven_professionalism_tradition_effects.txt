mark_elven_professionalism_tradition_as_discovered = {
	$TRADITION$ = {
		save_scope_as = tradition
	}

	if = {
		limit = {
			scope:tradition = culture_tradition:tradition_elven_professionalism
		}
		hidden_effect = {
			add_to_global_variable_list = {
				name = discovered_elven_professionalism
				target = flag:tradition_elven_professionalism
			}
		}
	}
}
